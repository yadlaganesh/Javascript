Function Statements & Expressions

	//function decloration does not give immediate result
	function WhatDoyou(job,name){
		return (name+'doing'+job);
	}
	console.log(WhatDoyou('dev','suresh'));


	//function expression will give immediate result
	var result = function(job, name){
		return (name+'doing'+job);
	}
	console.log(result('developer', 'ganesh'));
