Js Loops and Iteration

		var Names = ["Ganesh", "Suresh", "It", 1990, false, 'blue'];
		for(var i=0;i < Names.length;i++){
			console.log(Names[i]);
		}
		console.log('**********for loop end************');

		// while loop
		var i=0;
		while(i < Names.length){
			console.log(Names[i]);
			i++;
		}
		console.log('***********while loop end***********');

		//continue and break
		for(var i=0;i < Names.length;i++){
			if(typeof Names[i] !== 'string') {
				continue;
			}
			console.log(Names[i]);
		}
		for(var i=0;i < Names.length;i++){
			if(typeof Names[i] !== 'string') {
				break;
			}
			console.log(Names[i]);
		}
		console.log('***********continue and break***********');
		for(var i < Names.length - 1;i>Names.length;i--){
			console.log(i);
		}
		console.log('*********** Reverse loop ***********');
