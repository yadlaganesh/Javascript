			function tipCalculate(bill){
				if(bill < 50){
					return (bill*20/100);
				}else if(bill > 50 && bill < 200){
					return (bill*15/100);
				}else if(bill > 200){
					return (bill*10/100);
				}		
			}

			var tips = [tipCalculate(bills[0]),tipCalculate([bills[1]]),tipCalculate([bills[2]])];
