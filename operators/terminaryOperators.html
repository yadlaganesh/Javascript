Ternary Operator/ Switch

	var name = 'Ganesh';
	var age = 16;

	// ternary operator
	age >= 18 ? console.log(name+'is a boy'):console.log(name+'is a young boy');

	var drink = age >= 18 ? 'beer' : 'juice';
	console.log(drink);


	//switch operator
	var job = 'teacher';
	switch(job){ 
		case 'teacher' :
		console.log(name+' is a teacher');
		break;
		case 'designer' :
		console.log(name+' is a designer');
		break;
		case 'driver' :
		console.log(name+' is a driver');
		break;
		default:
			console.log(name+ ' is a does something...!');
	}
