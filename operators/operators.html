Operators

	

	// Math Operators
	var now = 2018;
	var SureshAge = now - 26;
	var NikeshAge = now - 22;
	console.log(SureshAge);
	console.log(NikeshAge);
	console.log(now * 2);
	console.log(now / 2);

	//Logical Operators
	var olderAge = SureshAge > NikeshAge;
	console.log(olderAge);
	console.log(typeof olderAge);

	// more operators
	x *= 2;
	console.log(x);
	x += 10;
	console.log(x);
	x++;
	console.log(x);
	--x;
	console.log(x);


	/****************************
		Operator Precedence

	*****************************/
	var yearGanesh = 1992;
	var fullAge = 18;
	var isFullAge = now - yearGanesh >= fullAge; // - has a high precidence and after >= 
	console.log(isFullAge);
	

	var average  = (SureshAge + NikeshAge) / 2; // grouping has high precidence
	console.log(average);
